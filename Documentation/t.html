Theme Configuration

Customize the Material-UI theme using the SCSS variables defined in the theme configuration file:

// src/themes/theme.js
import { createTheme } from '@mui/material/styles';

// Light Theme
export const lightTheme = createTheme({
  palette: {
    mode: 'light',
    primary: {
      light: '#e3f2fd',
      main: '#2196f3',
      dark: '#1e88e5',
      200: '#90caf9',
      800: '#1565c0'
    },
    secondary: {
      light: '#ede7f6',
      main: '#673ab7',
      dark: '#5e35b1',
      200: '#b39ddb',
      800: '#4527a0'
    },
    success: {
      light: '#b9f6ca',
      main: '#00e676',
      dark: '#00c853',
      200: '#69f0ae'
    },
    error: {
      light: '#ef9a9a',
      main: '#f44336',
      dark: '#c62828'
    },
    warning: {
      light: '#fff8e1',
      main: '#ffe57f',
      dark: '#ffc107'
    },
    orange: {
      light: '#fbe9e7',
      main: '#ffab91',
      dark: '#d84315'
    },
    grey: {
      50: '#f8fafc',
      100: '#eef2f6',
      200: '#e3e8ef',
      300: '#cdd5df',
      500: '#697586',
      600: '#4b5565',
      700: '#364152',
      900: '#121926'
    },
    background: {
      default: '#f8fafc',
      paper: '#ffffff'
    },
    text: {
      primary: '#121926',
      secondary: '#697586'
    }
  },
  typography: {
    fontFamily: '"Inter", "Roboto", "Helvetica", "Arial", sans-serif',
    h1: {
      fontSize: '2.5rem',
      fontWeight: 600,
    },
    // ... other typography settings
  },
  shape: {
    borderRadius: 8,
  },
});

// Dark Theme
export const darkTheme = createTheme({
  palette: {
    mode: 'dark',
    primary: {
      light: '#e3f2fd',
      main: '#2196f3',
      dark: '#1e88e5',
      200: '#90caf9',
      800: '#1565c0'
    },
    secondary: {
      light: '#d1c4e9',
      main: '#7c4dff',
      dark: '#651fff',
      200: '#b39ddb',
      800: '#6200ea'
    },
    background: {
      default: '#1a223f',
      paper: '#111936'
    },
    text: {
      primary: '#bdc8f0',
      secondary: '#8492c4',
      title: '#d7dcec'
    },
    dark: {
      level1: '#29314f',
      level2: '#212946'
    }
  },
  typography: {
    fontFamily: '"Inter", "Roboto", "Helvetica", "Arial", sans-serif',
    // ... dark theme typography
  }
});
Note: The theme configuration uses your actual SCSS color variables for consistency across the application. Make sure to update both the SCSS variables and the Material-UI theme when changing colors.
